					public class Animal
					{
						public string Name { get; set; }

						// 父類別的有參數建構子
						public Animal(string name)
						{
							this.Name = name;
							Console.WriteLine($"父類別：Animal {Name} 已建立");
						}
					}

					public class Dog : Animal
					{
						public string Breed { get; set; }

						// 子類別建構子：使用 : base(name) 將名字傳回給父類別
						public Dog(string name, string breed) : base(name) 
						{
							this.Breed = breed;
							Console.WriteLine($"子類別：品種為 {Breed} 的小狗已建立");
						}
					}
